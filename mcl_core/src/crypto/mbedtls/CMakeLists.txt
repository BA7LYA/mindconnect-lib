FIND_PATH(MBEDTLS_INCLUDE_PATH "mbedtls/version.h")
FIND_LIBRARY(MBEDTLS_LIBRARY "mbedtls")
FIND_LIBRARY(MBEDCRYPTO_LIBRARY "mbedcrypto")

MESSAGE(STATUS "mbedTLS Path: ${MBEDTLS_INCLUDE_PATH}")

SET(MCL_CRYPTO_INCLUDE_DIRECTORIES ${MBEDTLS_INCLUDE_PATH} PARENT_SCOPE)
SET(MCL_CRYPTO_LIBRARIES ${MBEDTLS_LIBRARY} ${MBEDCRYPTO_LIBRARY} PARENT_SCOPE)
SET(MCL_CRYPTO_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}/security_mbedtls.c" PARENT_SCOPE)
